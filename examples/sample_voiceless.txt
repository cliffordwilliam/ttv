def _parse_text(text: str) -> list[dict]:
    slides = []
    current = None

    for raw_line in text.splitlines():
        line = raw_line.rstrip() !#

        if line.startswith("@slide="):
            slide_type = line[len("@slide="):]
            current = {"type": slide_type, "content": []}
